🔄 STEP 7: MAPPER LAYER (🔥 MOST IMPORTANT)

AircraftMapper

Java
@Component
public class AircraftMapper {

    private final ObjectMapper objectMapper = new ObjectMapper();

    public AircraftDTO toDTO(AircraftEntity entity) {
        AircraftDTO dto = new AircraftDTO();

        dto.setId(entity.getId());
        dto.setRegistration(entity.getRegistration());
        dto.setModel(entity.getModel());
        dto.setManufacturer(entity.getManufacturer());
        dto.setCapacity(entity.getCapacity());
        dto.setStatus(entity.getStatus());
        dto.setYearBuilt(entity.getYearBuilt());
        dto.setLastMaintenance(entity.getLastMaintenance());
        dto.setNextMaintenance(entity.getNextMaintenance());

        try {
            CabinConfigDTO cabin =
                objectMapper.readValue(entity.getCabinConfig(), CabinConfigDTO.class);
            dto.setCabinConfig(cabin);
        } catch (Exception e) {
            throw new RuntimeException("Error parsing cabin config");
        }

        return dto;
    }
}

🧠 STEP 8: SERVICE LAYER (AGGREGATION LOGIC)

AdminDashboardService

Java
@Service
@RequiredArgsConstructor
public class AdminDashboardService {

    private final AircraftRepository aircraftRepository;
    private final AirportRepository airportRepository;
    private final CrewRepository crewRepository;

    private final AircraftMapper aircraftMapper;
    private final AirportMapper airportMapper;
    private final CrewMapper crewMapper;

    public AdminDashboardResponseDTO getDashboard() {

        List<AircraftDTO> aircraft =
            aircraftRepository.findAll()
                .stream()
                .map(aircraftMapper::toDTO)
                .toList();

        List<AirportDTO> airports = airportMapper.getAllAirports();

        List<CrewDTO> crew = crewMapper.getAllCrew();

        AdminStatsDTO stats = getStats();

        AdminDashboardResponseDTO response = new AdminDashboardResponseDTO();
        response.setAircraft(aircraft);
        response.setAirports(airports);
        response.setCrews(crew);
        response.setAdminStats(stats);

        return response;
    }
}
